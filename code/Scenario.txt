converting string to upper or lower case{
    function name = convertString(String pInputString, boolean checker){
        if checker is true{ //so if the user inputted parameter true
            pInputString = pInputString.toUpperCase()
        }
        else{
            pInputString = pInputString.toLowerCase()
        }
        return pInputString;
    }
}


Identify whether numeric value are in a given string{
    function name = numericFilter(String pInputString){
        char[] charArray = pInputString.toCharArray();
        int counter = 0;
        for(int i = 0; i < charArray.length; i ++){
            if(Character.isDigit(charArray[i])){
                counter ++;
            }
        }
        print "there are " + counter + " digits in your given string"
    }
}
